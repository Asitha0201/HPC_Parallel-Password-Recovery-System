# Run commands for all three phases

# Phase 1 - Serial
# Compile serial version and run it
gcc -O2 -lssl -lcrypto -o serial_crack serialProcessing.c
./serial_crack

# Phase 2 - OpenMP
# Compile OpenMP version and run it
gcc -O2 -fopenmp -o omp_crack omp_crack.c -lssl -lcrypto
./omp_crack                        # auto-detects cores
OMP_NUM_THREADS=8 ./omp_crack      # force 8 threads

# Phase 3 - MPI
# Compile MPI version and run it on 4 processes
mpicc -O2 -o mpi_crack mpi_crack.c -lssl -lcrypto
mpirun -np 4 ./mpi_crack
